.btn {
  margin-bottom: 1em;
  line-height: 1;
  border-radius: 0;
  border-width: 2px;
  font-weight: $font-weight-bold;
  padding: 0.75em 1em;
  min-height: 2.75em;

  text-decoration: none;

  @extend .hidden-print;

  &:hover,
  &:active,
  &:focus {
    text-decoration: underline;

    @include box-shadow(none);
  }

  &:focus {
    outline: 2px solid $yellow-light;
    border-color: $yellow-light;
  }

  &:after {
    line-height: inherit;
    position: static;
  }

  &.btn-default {
    border-color: $brand-primary;
    color: $brand-primary;
    background: none;

    &:hover {
      background: $brand-primary;
      color: #fff;
    }

    &:active,
    &:focus {
      background: #fff;
      color: $brand-primary;
    }
  }

  &.btn-primary {
    border-color: $brand-primary;

    &:hover,
    &:active {
      background: #fff;
      color: $brand-primary;
    }

    &:focus {
      color: #fff;
      background: $brand-primary;
    }
  }

  &.btn-secondary {
    border-color: #fff;
    background: #fff;
    color: $brand-primary;
  }

  &.btn-ctn {
    background: $yellow-light;
    color: #000;

    &:active {
      border-color: $text-color;

      outline-color: $text-color;

      font-size: 0.8em;

      @extend .icon--right;
    }
  }

  &.btn-menu {

    }

.btn-collapse {
  @extend .btn-default;

  span.toggling-text:after {
    content: '\00a0';
  }

  &:hover,
  &:focus {
    span.toggling-text,
    span.toggling-text:after {
      text-decoration: underline;
    }
  }

  .icon {
    margin-left: 0.3em;

    &:before {
      vertical-align: bottom;
    }
  }

  &:hover:not(:focus) {
    .icon:before {
      color: #fff;
    }
  }

  &.collapsed {
    &:span.icon {
      @include rotate(180deg);
    }
  }
}

.no-js {
  .btn-collapse {
    display: none;
    visibility: hidden;
  }
}
