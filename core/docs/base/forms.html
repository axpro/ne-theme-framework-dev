// Checkbox

// Please
// @see http://bootsnipp.com/snippets/featured/funky-radio-buttons
// @see http://webdesign.tutsplus.com/tutorials/quick-tip-easy-css3-checkboxes-and-radio-buttons--webdesign-8953

// Component
.checkbox {
  margin-bottom: 2px;
  margin-top: 0;

  label {
    color: $form-checkbox-color-unchecked;
    height: $form-control-height;
    line-height: $form-control-height;
    display: block;
    text-align: left;
    border: transparent solid 2px;
    border-radius: 0;
  }

  input {
    margin: 1px 0 0 13px;
  }

  // Checkbox.
  input[type='checkbox'] {
    height: $form-control-height;
    margin-left: -9999px;
    position: relative;
    float: left;

    // Default state.
    &:not(checked) + label {
      background-color: $form-checkbox-background-color-unchecked;
      padding-left: 2.778em;

      &:before {
        content: '';
        position: absolute;
        display: block;
        width: $form-checkbox-icon-size;
        height: $form-checkbox-icon-size;
        top: 0.8em;
        left: 1em;
        background: url('../images/svg/checkbox-unchecked.svg') no-repeat;
        background-size: auto $form-checkbox-icon-size;
      }
    }

    // Checked state.
    &:checked + label {
      color: $form-checkbox-color-checked;
      background-color: $form-checkbox-background-color-checked;

      &:before {
        background-image: url('../images/svg/checkbox-checked.svg');
      }
    }

    // Hover state, not checked.
    &:hover:not(checked) + label {
      color: $form-checkbox-color-unchecked-hover;
      border: $form-checkbox-border-color-unchecked-hover solid 2px;
      background-color: $form-checkbox-background-color-unchecked-hover;
    }

    // Hover state, checked.
    &:hover:checked + label {
      color: $form-checkbox-color-checked-hover;
      border: $form-checkbox-border-color-checked-hover solid 2px;
      background-color: $form-checkbox-background-color-checked-hover;
    }
  }
}

// Turning off completely custom checkboxes when SVG's are not available
.no-svg {
  .checkbox {
    label {
      height: auto;
      line-height: inherit;
      display: inline;
      background: none;
      padding-left: 0;
    }

    input[type='checkbox'] {
      margin: 0;
      float: none;
      height: auto;
      &:before {
        display: none;
      }

      &:not(checked) + label {
        background: none;
        padding: 0;
        &:before {
          display: none;
        }
      }

      &:checked + label {
        color: inherit;
        background: none;

        &:before {
          display: none;
        }
      }
    }
  }
}

//
.select {
  /* Hide the arrow */
  appearance: none;
  -moz-appearance: none; /* Firefox */
  -webkit-appearance: none; /* Safari and Chrome */
  text-indent: 0.01px;
  text-overflow: '';
  cursor: pointer;

  // Fix for IE 10+.
  &::-ms-expand {
    display: none;
  }

  &.form-control {
    @include select-list-arrow($form-control-border-color);
    // To not make it overlap.
    padding-right: 50px;
  }

  // Hover state.
  &:hover {
    border-color: $form-control-border-select-color-hover;

    @include select-list-arrow($form-control-border-select-color-hover);
  }

  // Down state.
  &.is-down {
    @include select-list-arrow(#ffffff, true);
  }

}

select {
  @extend .select;
}
